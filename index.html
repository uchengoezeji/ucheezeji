                    </p>

                    <p class="font-sans mb-4">

They are a cultural worker, specializing in creative non-fiction, essays, poetry, and curriculum curation.

                    </p>

                    <p class="font-sans mb-4">

                        Their purpose derives from creating and fostering environments for reflective critical dialogue aimed at healing and transforming society.

                    </p>

                    <p class="font-sans">

                        Uche's work embraces the Black feminist belief that the personal is always political and is rooted in a practice of love, exploration of pleasure, and embodied political perspectives.

    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const dockItems = document.querySelectorAll('.dock-item');

            const windows = document.querySelectorAll('.window');

            const closeButtons = document.querySelectorAll('.close-btn');

            const contactForm = document.getElementById('contact-form');

            const contactStatus = document.getElementById('contact-status');


            // Map to store current Z-index for window stacking

            const windowZIndex = {};

            let nextZIndex = 11;


            // Function to bring a clicked window to the front

            const bringToFront = (windowElement) => {

                if (windowZIndex[windowElement.id] !== nextZIndex) {

                    windowElement.style.zIndex = nextZIndex;

                    windowZIndex[windowElement.id] = nextZIndex;

                    nextZIndex++;

                }

            };


            // Event listeners for dock items

            dockItems.forEach(item => {

                item.addEventListener('click', (e) => {

                    e.preventDefault();

                    const targetId = item.dataset.target;

                    const targetWindow = document.getElementById(targetId);

                    

                    if (targetWindow) {

                        // Check if the window is already open

                        if (targetWindow.classList.contains('hidden')) {

                            targetWindow.classList.remove('hidden');

                        }

                        // Bring the window to the front

                        bringToFront(targetWindow);

                    }

                });

            });


            // Event listeners for close buttons

            closeButtons.forEach(btn => {

                btn.addEventListener('click', (e) => {

                    e.stopPropagation();

                    const targetId = btn.dataset.target;

                    const targetWindow = document.getElementById(targetId);

                    if (targetWindow) {

                        targetWindow.classList.add('hidden');

                    }

                });

            });


            // Make windows draggable

            windows.forEach(windowElement => {

                let isDragging = false;

                let offsetX, offsetY;


                const startDrag = (e) => {

                    isDragging = true;

                    offsetX = e.clientX - windowElement.getBoundingClientRect().left;

                    offsetY = e.clientY - windowElement.getBoundingClientRect().top;

                    windowElement.style.cursor = 'grabbing';

                    bringToFront(windowElement);

                };


                const drag = (e) => {

                    if (!isDragging) return;

                    e.preventDefault();

                    const newX = e.clientX - offsetX;

                    const newY = e.clientY - offsetY;

                    windowElement.style.left = `${newX}px`;

                    windowElement.style.top = `${newY}px`;

                };


                const endDrag = () => {

                    isDragging = false;

                    windowElement.style.cursor = 'grab';

                };


                // Add event listeners for both mouse and touch

                windowElement.querySelector('.window-header').addEventListener('mousedown', startDrag);

                document.addEventListener('mousemove', drag);

                document.addEventListener('mouseup', endDrag);

                windowElement.querySelector('.window-header').addEventListener('touchstart', (e) => startDrag(e.touches[0]));

                document.addEventListener('touchmove', (e) => drag(e.touches[0]));

                document.addEventListener('touchend', endDrag);


                // Add event listener to bring window to front on click

                windowElement.addEventListener('mousedown', () => bringToFront(windowElement));

            });


            // Contact form submission handler

            contactForm.addEventListener('submit', (e) => {

                e.preventDefault();

                // In a real application, you would send this data to a backend.

                // For this portfolio, we just show a success message.

                contactForm.classList.add('hidden');

                contactStatus.classList.remove('hidden');

                

                // Reset the form after a short delay to allow for reuse

                setTimeout(() => {

                    contactForm.reset();

                    contactForm.classList.remove('hidden');

                    contactStatus.classList.add('hidden');

                }, 3000); // Hide the message after 3 seconds

            });


            // Typing animation script

            const typingTextElement = document.getElementById('typing-text');

            const phrases = [

                "// INITIATING_BLACK_FEMINIST_NARRATIVE...",

                "// GENERATING_PAN_AFRICAN_FUTURES...",

                "// CONNECTING_TO_GLOBAL_AFRICAN_NETWORK...",

                "// ACCESSING_INTIMATES_ARCHIVES..."

            ];

            let phraseIndex = 0;

            let charIndex = 0;

            let isDeleting = false;


            function type() {

                const currentPhrase = phrases[phraseIndex];

                const displayedText = isDeleting

                    ? currentPhrase.substring(0, charIndex - 1)

                    : currentPhrase.substring(0, charIndex + 1);


                typingTextElement.textContent = displayedText;


                if (!isDeleting && charIndex === currentPhrase.length) {

                    isDeleting = true;

                    setTimeout(type, 1500); // Pause at end of phrase

                } else if (isDeleting && charIndex === 0) {

                    isDeleting = false;

                    phraseIndex = (phraseIndex + 1) % phrases.length;

                    setTimeout(type, 500); // Pause before typing next phrase

                } else {

                    const typingSpeed = isDeleting ? 40 : 80; // Speed of deleting vs typing

                    setTimeout(type, typingSpeed);

                }

                

                charIndex = isDeleting ? charIndex - 1 : charIndex + 1;

            }


            type();

        });


    </script>
